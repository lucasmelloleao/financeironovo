create or replace view comercial_pro.fin_decsaldo_v as
select distinct
b.cd_organiz, b.cd_empresa, b.nr_processo,i.nr_seqfat,
b.cd_parceiro, b.dt_movto,b.nr_nota,
b.tp_fatura, i.dt_juros, i.dt_vencimento, b.cd_tipodocto, tp_moeda,
CAST
((select sum(vl_operacao) from
fin_movto
where
cd_empresa = i.cd_empresa
and
nr_processo = i.nr_processo
and
nr_seqfat = i.nr_seqfat)
AS NUMBER(13,2)) vl_saldo
from
fin_base b
inner
join fin_item i on
b.cd_empresa = i.cd_empresa
and
b.nr_processo = i.nr_processo
inner
join fin_movto m on
m.nr_processo = i.nr_processo
and
m.nr_seqfat = i.nr_seqfat
and
 m.cd_Empresa = i.cd_empresa
and
exists
(
select sum(vl_operacao) from
fin_movto
where
cd_empresa = i.cd_empresa and
nr_processo = i.nr_processo
and
nr_seqfat = i.nr_seqfat
group
by b.nr_processo,i.nr_seqfat having
sum
(vl_operacao) > 0);

CREATE OR REPLACE TRIGGER COMERCIAL_PRO.SALDO_CAIXA
AFTER INSERT
   ON FIN_LANCAMENTO
   FOR EACH ROW

DECLARE
   v_CONT number(1);

BEGIN

   SELECT COUNT(CD_EMPRESA) into V_CONT FROM FIN_SALDO WHERE cd_organiz = :new.cd_organiz and cd_empresa = :new.cd_empresa
   and tp_pagamento = :new.tp_pagamento;
   if (v_cont > 0) then


   UPDATE FIN_SALDO SET VL_SALDO = :NEW.VL_OPERACAO + VL_SALDO
   WHERE CD_ORGANIZ = :NEW.CD_ORGANIZ AND CD_EMPRESA = :NEW.CD_EMPRESA
   AND TP_PAGAMENTO = :NEW.TP_PAGAMENTO;
   else

   insert into fin_saldo
   ( cd_organiz, cD_empresa, vl_saldo, tp_pagamento )
       values (:new.cd_organiz, :new.cd_empresa, :new.vl_operacao, :new.tp_pagamento);
   end if;

END;
